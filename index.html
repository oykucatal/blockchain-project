<body>

  <h1>Counter dApp</h1>

  <button id="connect">Connect Wallet</button>

  <button id="get">Get Value</button>

  <button id="inc">Increment</button>


  <p>Value: <span id="value">-</span></p>


<script>

  const CONTRACT_ADDRESS = "BURAYA-CONTRACT-ADRESİNİ_YAZ";

  const ABI = BURAYA_ABI_YAPISTIR;  /* JSON biçiminde, tırnaklar ve köşeli parantez ile */


  let provider;

  let contract;


  document.getElementById("connect").onclick = async () => {

    await window.ethereum.request({ method: "eth_requestAccounts" });

    provider = new ethers.providers.Web3Provider(window.ethereum);

    contract = new ethers.Contract(CONTRACT_ADDRESS, ABI, provider.getSigner());

    alert("Wallet connected!");

  };


  document.getElementById("get").onclick = async () => {

    const v = await contract.count();

    document.getElementById("value").innerText = v.toString();

  };


  document.getElementById("inc").onclick = async () => {

    const tx = await contract.increment();

    await tx.wait();

    alert("Incremented!");

  };

</script>
